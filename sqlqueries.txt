create database playpal_db;
use playpal_db;

create table User(
user_id int auto_increment primary key,
name varchar(30) NOT NULL,
dob date,
age int NOT NULL,
email varchar(40),
address varchar(100),
weight numeric(5,2),
height int,
username varchar(20) NOT NULL UNIQUE,
password varchar(20) NOT NULL,
bmi numeric(4,2)),
event_id int,
foreign key(event_id) references Event);

create table Event(
event_id int auto_increment primary key,
event_name varchar(100) NOT NULL UNIQUE,
sport varchar(100),
ageLimit int,
venue varchar(255),
time time,
date date,
noOfParticipants int);

create Event_creator(
event_id int primary key,
user_id int,
foreign key(event_id) references Event,
foreign key(user_id) references User, 
);

create table Tournament(
tour_id int auto_increement primary key,
noOfTeams int,
tour_name varchar(255) NOT NULL UNIQUE,
entryFee numeric(5,2),
tour_ageLimit int,
tour_venue varchar(20),
tour_time varchar(20),
tour_date varchar(20),
tour_noOfParticipants int);


create table Team(
team_id int auto_increement primary key,
noOfPlayers int,
teamName varchar(50),
);

create table tour_participation(
user_id int,
tour_id int,
team_id int,
foreign key(user_id) references User,
foreign key(tour_id) references Tournament,
foreign key(team_id) references Team,
primary key(user_id, tour_id);
);


create table equipment(
equip_id int auto_increement,
userid_owner int,
userid_rentee int,
EquipName varchar(50) NOT NULL,
address varchar(100),
rentamt float,
condition varchar(50),
primary key(equip_id),
foriegn key(userid_owner) references User(user_id),
foreign key(userid_rentee) references User(user_id));
